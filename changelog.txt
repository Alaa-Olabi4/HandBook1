// script.js :
// ========================
// Persistent Expiration Check
// ========================
const EXPIRATION_KEY = 'app_expiration';
const expirationDate = new Date('2025-12-31T23:59:59');

// Check if we've previously recorded expiration
if (localStorage.getItem(EXPIRATION_KEY)) {
    showExpiration();
    throw new Error("Application expired");
}

// Check current date
const currentDate = new Date();
if (currentDate > expirationDate) {
    // Set persistent flag
    localStorage.setItem(EXPIRATION_KEY, 'true');
    
    // Clear other data
    const preserveKeys = [EXPIRATION_KEY];
    Object.keys(localStorage).forEach(key => {
        if (!preserveKeys.includes(key)) {
            localStorage.removeItem(key);
        }
    });
    
    showExpiration();
    throw new Error("Application expired");
}

function showExpiration() {
    document.body.innerHTML = `
        <div style="display: flex; justify-content: center; align-items: center; height: 100vh; text-align: center; padding: 20px; flex-direction: column; background: linear-gradient(135deg, #6a11cb 0%, #2575fc 100%); font-family: 'Poppins', sans-serif; color: white;">
            <div class="logo" style="margin-bottom: 30px;">
                <i class="fas fa-blog" style="font-size: 50px; margin-bottom: 15px;"></i>
                <div style="font-size: 24px; font-weight: 700;color: #ffcc00;">Hand<span style="color: #ffcc00;">Book</span></div>
            </div>
            <h1 style="font-size: 28px; margin-bottom: 20px; font-weight: 600;">Content Expired</h1>
            <p style="font-size: 18px; margin-bottom: 30px; max-width: 500px; line-height: 1.6;">
                This application's content has expired .
            </p>
            <p style="font-size: 16px; opacity: 0.9;">
                Please contact support for more information.
            </p>
        </div>
    `;
}
